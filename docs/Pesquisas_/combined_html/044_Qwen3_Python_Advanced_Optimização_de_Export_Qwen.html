    def basic_markdown_processing(self, text):

            elif line.strip():

    def basic_markdown_processing(self, text):
        lines = text.split('\n')
        html_lines = []
        for line in lines:
            if line.startswith('# '):
                html_lines.append(f'<h1>{line[2:]}</h1>')
            elif line.startswith('## '):
                html_lines.append(f'<h2>{line[3:]}</h2>')
            elif line.startswith('### '):
                html_lines.append(f'<h3>{line[4:]}</h3>')
            elif 'üë§ Voc√™' in line or 'ü§ñ Qwen AI' in line:
                if 'üë§ Voc√™' in line:
                    html_lines.append(f'<div class="message user"><strong>{line}</strong></div>')
                else:
                    html_lines.append(f'<div class="message assistant"><strong>{line}</strong></div>')
            elif line.strip().startswith('```'):
                html_lines.append('<pre><code>')
            elif line.strip() == '```':
                html_lines.append('</code></pre>')
            elif line.strip() == '---':
                html_lines.append('<hr>')
            elif line.strip():
                html_lines.append(f'<p>{line}</p>')
            else:
                html_lines.append('<br>')
        return '\n'.join(html_lines)

    def html_to_pdf_selenium(self, html_content, pdf_path):
        driver = None

            driver = self.setup_chrome_driver()

            with tempfile.NamedTemporaryFile(mode='w', suffix='.html', delete=False, encoding='utf-8') as f:
                f.write(html_content)

            driver.get(f'file://{temp_html}')
            WebDriverWait(driver, 10).until(
                lambda d: d.execute_script('return document.readyState') == 'complete'
            )
            print_options = {
                'landscape': False,
                'displayHeaderFooter': False,
                'printBackground': True,
                'preferCSSPageSize': True,
                'paperWidth': 8.27,

                'marginTop': 0.4,
                'marginBottom': 0.4,
                'marginLeft': 0.4,
                'marginRight': 0.4,
                'scale': 1.0
            }
            result = driver.execute_cdp_cmd('Page.printToPDF', print_options)
            pdf_data = base64.b64decode(result['data'])
            with open(pdf_path, 'wb') as f:

            os.unlink(temp_html)
            return True

            if driver:

            return False
        finally:
            if driver:
